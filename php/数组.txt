<?php
//目标：删除一维或多维数组中的元素，unset方法
$n = count($arr);			//如果把count($arr)直接放for里，得出的结果就是错的。
//经检验，当数组单元数目在循环中会动态改变时，不能采用实时计算的方法count($arr)，这样$i的最大值会实时发生改变，就达不到需要的效果了。
for ($i=0;$i<$n;$i++) {
	if ($arr[$i]["id"] == $obj) {
		unset($arr[$i]);	//unset不会改变数据的下标
	}
}
sort($arr);				//由于unset不会改变数组的下标，所以需要重新整理，但这样的结果，改变了原排序，如果需要原排序效果，需要单独处理
?>
<?php
