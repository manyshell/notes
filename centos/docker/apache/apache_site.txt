<IfModule mod_ssl.c>
<VirtualHost *:443>
	ServerName cdn.sylinzi.com
        ServerAdmin wildwindfeng@gmail.com
        DocumentRoot /var/www/repo

        #   SSL Engine Switch:
        #   Enable/Disable SSL for this virtual host.
        SSLEngine on

        #   A self-signed (snakeoil) certificate can be created by installing
        #   the ssl-cert package. See
        #   /usr/share/doc/apache2.2-common/README.Debian.gz for more info.
        #   If both key and certificate are stored in the same file, only the
        #   SSLCertificateFile directive is needed.
        SSLCertificateFile /etc/sslmate/*.sylinzi.com.crt
        SSLCertificateKeyFile /etc/sslmate/*.sylinzi.com.key
	SSLCertificateChainFile /etc/sslmate/*.sylinzi.com.chain.crt
</VirtualHost>
</IfModule>
