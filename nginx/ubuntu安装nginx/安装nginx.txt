由于之前系统安装过apache2，所以需要先卸载
apt-get purge apache2
apt-get purge apache2-dev

安装nginx
apt-get install nginx

[root@localhost html]# nginx -V
nginx version: nginx/1.4.6 (Ubuntu)

# netstat -tulnp | grep 80
tcp        0      0 0.0.0.0:80              0.0.0.0:*               LISTEN      356/nginx       
tcp6       0      0 :::80                   :::*                    LISTEN      356/nginx       
启动成功了

默认配置文件
/etc/nginx/nginx.conf

默认网页目录
/usr/share/nginx/html

默认站点配置文件
/etc/nginx/sites-enabled/default
---
listen       80;                                    //监听80端口
location / {
    index  index.php index.html index.htm;          //默认index文件
    limit_rate 50k;                                 //每个连接限速50k
}
location ~ \.php$ {
    fastcgi_pass   127.0.0.1:9000;
    fastcgi_index  index.php;
    fastcgi_param  SCRIPT_FILENAME  $document_root$fastcgi_script_name;
    include        fastcgi_params;
}

---

重启nginx
/usr/sbin/nginx -s reload
